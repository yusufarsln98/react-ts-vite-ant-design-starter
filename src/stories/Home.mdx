- Experiment with props and controls to customize the look and feel.

## Template Structure

```typescript
import type { Meta, StoryFn } from '@storybook/react'
import { ComponentType } from 'react'
import {
	SomeComponent,
	SomeComponentProps,
} from '@/components/some-component'

export default {
	title: 'SomeComponent',
	component: SomeComponent,
	argTypes: {
		placeholder: { control: 'text' },
		disabled: { control: 'boolean' },
	},
	tags: ['autodocs'],
} as Meta

const Template: StoryFn<SomeComponentProps> = (args) => (
	<SomeComponent {...args} />
)

export const Default = Template.bind({})
Default.args = {
	placeholder: 'Enter text...',
	dummyProp: 'Default value',
}

export const Disabled = Template.bind({})
Disabled.args = {
	placeholder: 'Disabled...',
	disabled: true,
	dummyProp: 'Disabled state',
}

export const Dummy = Template.bind({})
ReadOnly.args = {
	dummyProp: 'Some state',
}
```
